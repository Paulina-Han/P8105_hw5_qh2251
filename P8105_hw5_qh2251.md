      disposition == "Open/No arrest" ~"unsolved",
      disposition == "Closed by arrest" ~"solved"
    )
  ) %>% 
  relocate(city_state) %>% 
   filter(city_state != "Tulsa,AL")

#resolution of crimes
df_summary = df %>% 
  group_by(city_state) %>% 
  summarize(
    unsolved = sum(resolution == "unsolved"),
    n=n()
  )
